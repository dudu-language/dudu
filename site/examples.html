Payload enums and match

enum Message:
    Move(i32, i32)
    Write(str)
    Quit

def handle(message: &const[Message]):
    match message:
        Message.Move(x, y): print(x, y)
        Message.Write(text): print(text)
        Message.Quit: print("done")
